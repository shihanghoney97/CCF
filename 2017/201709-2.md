#include <iostream>

#include <algorithm>

    int w, s, c, e;
}t;

using namespace std;

int main()
{
    int n, k;
    cin >> n >> k;
    int key[n];
    for(int i=0; i<n; i++)
    {
        key[i] = i+1;
    }
    t a[k];
    int max = 20000, min = 0;
    for(int i=0; i<k; i++)
    {
        cin >> a[i].w >> a[i].s >> a[i].c;
        a[i].e = a[i].s + a[i].c;
        min = a[i].s<min?a[i].s:min;
        max = a[i].e>max?a[i].e:max;
    }

    for(int i=min; i<=max; i++)
    {
        vector<int> ret;
        for(int j=0; j<k; j++)
        {
            if(a[j].e==i)
            {
                ret.push_back(a[j].w);
            }
        }
        sort(ret.begin(), ret.end());
        int p = 0, num = 0;;
        while(num<ret.size())
        {
            while(p<n)
            {
                if(key[p]==0)
                {
                    key[p] = ret[num];
                    num++;
                    break;
                }
                p++;
            }
        }
        for(int j=0; j<k; j++)
        {
            if(a[j].s==i)
            {
                for(int q=0; q<n; q++)
                {
                    if(key[q]==a[j].w)
                    {
                        key[q] = 0;
                    }
                }
            }
        }
    }

    for(int i=0; i<n; i++)
    {
        cout << key[i] << ' ';
    }
    return 0;
}
